### 通用导航栏组件，根据配置控制菜单的显隐
```
{
    path: "/hide",
    name: "隐藏页面",
    component: AdminView,
    meta: {
      hideInMenu: true,
    },
},
```
```
const visibleRoutes = routes.filter((item, index) => {
  if (item.meta?.hideInMenu) {
    return false;
  }
  return true;
});

<a-menu-item v-for="item in visibleRoutes" :key="item.path">
    {{item.name}}
</a-menu-item>
```
